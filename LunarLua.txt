local function NormalizeAnimationId(id)
	if not id then
		return nil
	end

	id = tostring(id)

	if id:find("rbxassetid://") then
		return id
	end

	local numberId = id:match("%d+")
	if numberId then
		return "rbxassetid://" .. numberId
	end

	return nil
end

local function CreateGrayDummy(world)
	local description = Instance.new("HumanoidDescription")

	local dummy
	local success = pcall(function()
		dummy = Players:CreateHumanoidModelFromDescription(description, Enum.HumanoidRigType.R15)
	end)

	if not success or not dummy then
		return nil
	end

	dummy.Name = "GrayPreviewDummy"

	for _, item in ipairs(dummy:GetDescendants()) do
		if item:IsA("Script") or item:IsA("LocalScript") or item:IsA("ModuleScript") then
			item:Destroy()
		elseif item:IsA("Accessory") or item:IsA("Shirt") or item:IsA("Pants") or item:IsA("ShirtGraphic") then
			item:Destroy()
		elseif item:IsA("BasePart") then
			item.Anchored = false
			item.CanCollide = false
			item.CastShadow = false
			item.Color = Color3.fromRGB(135, 135, 135)
			item.Material = Enum.Material.SmoothPlastic
		elseif item:IsA("Decal") or item:IsA("Texture") then
			item.Transparency = 1
		end
	end

	local humanoid = dummy:FindFirstChildOfClass("Humanoid")
	if humanoid then
		humanoid.DisplayDistanceType = Enum.HumanoidDisplayDistanceType.None
		humanoid.HealthDisplayType = Enum.HumanoidHealthDisplayType.AlwaysOff
	end

	dummy.Parent = world
	dummy:PivotTo(CFrame.new(0, 0, 0) * CFrame.Angles(0, math.rad(180), 0))

	return dummy
end

local function PlayPreviewAnimation(dummy, animationId, label, text, seconds)
	if not dummy or not dummy.Parent then
		return
	end

	local humanoid = dummy:FindFirstChildOfClass("Humanoid")
	if not humanoid then
		return
	end

	local animator = humanoid:FindFirstChildOfClass("Animator")
	if not animator then
		animator = Instance.new("Animator")
		animator.Parent = humanoid
	end

	if label then
		label.Text = text or "Preview"
	end

	local normalizedId = NormalizeAnimationId(animationId)
	if not normalizedId then
		task.wait(seconds or 2)
		return
	end

	local animation = Instance.new("Animation")
	animation.AnimationId = normalizedId

	local track
	local ok = pcall(function()
		track = animator:LoadAnimation(animation)
	end)

	animation:Destroy()

	if ok and track then
		track.Looped = true
		track.Priority = Enum.AnimationPriority.Action
		track:Play(0.15)

		task.wait(seconds or 2)

		pcall(function()
			track:Stop(0.15)
			track:Destroy()
		end)
	else
		task.wait(seconds or 2)
	end
end

local function MakeNotificationPullable(notif)
	local dragging = false
	local dragStart = nil
	local originalPosition = nil
	local returnTween = nil

	local maxPullX = 18
	local maxPullY = 10
	local pullStrength = 0.12

	local function clamp(value, minValue, maxValue)
		return math.max(minValue, math.min(maxValue, value))
	end

	notif.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			originalPosition = notif.Position

			if returnTween then
				returnTween:Cancel()
				returnTween = nil
			end

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false

					if notif and notif.Parent and not notif:GetAttribute("Closing") then
						returnTween = Tween(
							notif,
							TweenInfo.new(0.32, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
							{ Position = originalPosition }
						)
					end
				end
			end)
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and notif and notif.Parent and originalPosition then
			if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
				local delta = input.Position - dragStart

				local pullX = clamp(delta.X * pullStrength, -maxPullX, maxPullX)
				local pullY = clamp(delta.Y * pullStrength, -maxPullY, maxPullY)

				notif.Position = UDim2.new(
					originalPosition.X.Scale,
					originalPosition.X.Offset + pullX,
					originalPosition.Y.Scale,
					originalPosition.Y.Offset + pullY
				)
			end
		end
	end)
end

function Library:Notify(message, color, icon)
	IncomingNotification = true

	task.spawn(function()
		message = tostring(message)

		local notif = NotificationSample:Clone()
		notif.Parent = ScreenGui
		notif.Position = UDim2.fromScale(0.5, -0.1)
		notif.UIScale.Scale = 0.5
		notif.Visible = true
		notif.Name = message
		notif:SetAttribute("close", false)
		notif:SetAttribute("Closing", false)

		MakeNotificationPullable(notif)

		if color and typeof(color) == "Color3" then
			notif.UIStroke.Color = color
			notif.ImageLabel.ImageColor3 = color

			local closeStroke = notif.Close:FindFirstChildOfClass("UIStroke")
			if closeStroke then
				closeStroke.Color = color
			end
		end

		if icon then
			if NotificationIcons[icon] then
				notif.ImageLabel.Image = NotificationIcons[icon]
			elseif tonumber(icon) then
				notif.ImageLabel.Image = "rbxassetid://" .. tostring(tonumber(icon))
			else
				notif.ImageLabel.Image = tostring(icon)
			end
		end

		notif.TextLabel.MaxVisibleGraphemes = 0
		notif.TextLabel.Text = message

		Tween(notif, TweenInfo.new(0.7, Enum.EasingStyle.Cubic, Enum.EasingDirection.Out), {
			Position = UDim2.new(0.5, 0, 0, 10)
		})

		Tween(notif.UIScale, TweenInfo.new(0.8, Enum.EasingStyle.Cubic, Enum.EasingDirection.Out), {
			Scale = 0.8
		})

		Tween(notif.TextLabel, TweenInfo.new(0.7, Enum.EasingStyle.Cubic, Enum.EasingDirection.Out), {
			MaxVisibleGraphemes = #message
		})

		notif.Close.MouseButton1Click:Connect(function()
			notif:SetAttribute("close", true)
		end)

		task.wait()
		IncomingNotification = false

		local startClock = os.clock()

		repeat
			task.wait()
		until os.clock() - startClock > 5 or IncomingNotification or notif:GetAttribute("close")

		if notif and notif.Parent then
			notif:SetAttribute("Closing", true)

			local finish = Tween(notif, TweenInfo.new(0.3, Enum.EasingStyle.Cubic, Enum.EasingDirection.Out), {
				Position = UDim2.fromScale(0.5, -0.1)
			})

			finish.Completed:Connect(function()
				if notif then
					notif:Destroy()
				end
			end)
		end
	end)
end

local function FetchAllAnimationPacks()
	local success, result = pcall(function()
		local jsonContent = game:HttpGet("https://raw.githubusercontent.com/7yd7/sniper-Emote/refs/heads/test/AnimationSniper.json")
		local decoded = HttpService:JSONDecode(jsonContent)
		return decoded.data or decoded or {}
	end)

	if not success or type(result) ~= "table" then
		return FallbackAnimations()
	end

	local packs = {}

	for _, item in ipairs(result) do
		local itemName = item.name or item.Name
		local itemId = item.id or item.Id
		local bundledItems = item.bundledItems or item.BundledItems

		if itemId and itemName and bundledItems then
			table.insert(packs, {
				Name = tostring(itemName),
				Id = tonumber(itemId) or tostring(itemId),
				Description = "Preview and apply the " .. tostring(itemName) .. " animation pack.",
				BundledItems = bundledItems
			})
		end
	end

	table.sort(packs, function(a, b)
		return a.Name < b.Name
	end)

	if #packs == 0 then
		return FallbackAnimations()
	end

	return packs
end

local function GuessCategoryFromName(name)
	name = string.lower(tostring(name or ""))

	if name:find("swimidle") or name:find("swim idle") then
		return "swimidle"
	elseif name:find("idle") then
		return "idle"
	elseif name:find("walk") then
		return "walk"
	elseif name:find("run") then
		return "run"
	elseif name:find("jump") then
		return "jump"
	elseif name:find("fall") then
		return "fall"
	elseif name:find("climb") then
		return "climb"
	elseif name:find("swim") then
		return "swim"
	end

	return nil
end

local function GuessCategoryFromIndex(index)
	local order = {"climb", "fall", "idle", "jump", "run", "swim", "walk"}
	return order[index]
end

local function ExtractAssetId(value)
	if typeof(value) == "table" then
		return value.id or value.Id or value.assetId or value.AssetId or value.animationId or value.AnimationId
	end

	return value
end

local function ResolveAnimationMappings(itemData)
	local cacheKey = tostring(itemData.Id or itemData.Name or "Unknown")

	if AnimationCache[cacheKey] then
		return AnimationCache[cacheKey]
	end

	local mappings = {}

	local function addMapping(category, name, animationId)
		if not category or not animationId then
			return
		end

		local normalized = NormalizeAnimationId(animationId)
		if not normalized then
			return
		end

		table.insert(mappings, {
			category = string.lower(tostring(category)),
			name = tostring(name or category),
			animationId = normalized
		})
	end

	for _, assetIds in pairs(itemData.BundledItems or {}) do
		local list = {}

		if type(assetIds) == "table" then
			for _, value in pairs(assetIds) do
				table.insert(list, ExtractAssetId(value))
			end
		else
			table.insert(list, ExtractAssetId(assetIds))
		end

		for index, assetId in ipairs(list) do
			local cleanId = tostring(assetId or ""):match("%d+")

			if cleanId then
				local foundRealMappings = false

				local success, objects = pcall(function()
					return game:GetObjects("rbxassetid://" .. cleanId)
				end)

				if success and objects then
					local function searchTree(parent, parentPath)
						for _, child in ipairs(parent:GetChildren()) do
							if child:IsA("Animation") then
								local animationPath = parentPath .. "." .. child.Name
								local pathParts = animationPath:split(".")

								local category = pathParts[#pathParts - 1]
								local name = pathParts[#pathParts]

								if category and child.AnimationId and child.AnimationId ~= "" then
									foundRealMappings = true
									addMapping(category, name, child.AnimationId)
								end
							elseif #child:GetChildren() > 0 then
								searchTree(child, parentPath .. "." .. child.Name)
							end
						end
					end

					for _, obj in ipairs(objects) do
						searchTree(obj, obj.Name)

						obj.Parent = workspace
						task.delay(1, function()
							if obj then
								obj:Destroy()
							end
						end)
					end
				end

				if not foundRealMappings then
					local category = nil

					pcall(function()
						local info = MarketplaceService:GetProductInfo(tonumber(cleanId))
						category = GuessCategoryFromName(info.Name)
					end)

					category = category or GuessCategoryFromIndex(index)

					if category then
						addMapping(category, category, "rbxassetid://" .. cleanId)
					end
				end
			end
		end
	end

	AnimationCache[cacheKey] = mappings
	return mappings
end

local function ApplyAnimationPack(item)
	local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
	local humanoid = character:FindFirstChildOfClass("Humanoid")
	local animate = character:FindFirstChild("Animate")

	if not humanoid then
		Library:Notify("Humanoid not found.", Color3.fromRGB(255, 100, 100), "x")
		return
	end

	if humanoid.RigType ~= Enum.HumanoidRigType.R15 then
		Library:Notify("Animation catalog needs R15.", Color3.fromRGB(255, 100, 100), "x")
		return
	end

	if not animate then
		Library:Notify("Animate script not found.", Color3.fromRGB(255, 100, 100), "x")
		return
	end

	local mappings = ResolveAnimationMappings(item)

	if #mappings == 0 then
		Library:Notify("No animation IDs found.", Color3.fromRGB(255, 100, 100), "x")
		return
	end

	local animator = humanoid:FindFirstChildOfClass("Animator")
	if animator then
		for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
			pcall(function()
				track:Stop(0.12)
			end)
		end
	end

	for _, mapping in ipairs(mappings) do
		if mapping.category and mapping.animationId then
			local folder = animate:FindFirstChild(mapping.category)

			if folder then
				local exact = folder:FindFirstChild(mapping.name)

				if exact and exact:IsA("Animation") then
					exact.AnimationId = mapping.animationId
				else
					for _, animObj in ipairs(folder:GetChildren()) do
						if animObj:IsA("Animation") then
							animObj.AnimationId = mapping.animationId
						end
					end
				end
			end
		end
	end

	CurrentAnimationData = item

	animate.Disabled = true
	task.wait(0.08)
	animate.Disabled = false

	humanoid.Jump = false

	pcall(function()
		humanoid:ChangeState(Enum.HumanoidStateType.Running)
	end)

	Library:Notify(item.Name .. " applied.", Color3.fromRGB(255, 255, 255), "check")
end

LocalPlayer.CharacterAdded:Connect(function(character)
	if CurrentAnimationData then
		task.wait(1.25)

		if character and character.Parent then
			ApplyAnimationPack(CurrentAnimationData)
		end
	end
end)

	local SlotAngles = {-72, -36, 0, 36, 72}
	local Radius = 116

	local function GetMaxScrollOffset()
		return math.max(0, #dock.Tabs - dock.VisibleTabs)
	end

	local function GetTabPositionByAngle(angle)
		local radians = math.rad(angle)
		local x = OrbCenterX + math.cos(radians) * Radius
		local y = math.sin(radians) * Radius
		return UDim2.new(0, x, 0.5, y)
	end

	local function ForceActiveTab(tab)
		if not tab or not tab.Button then
			return
		end

		tab.Button.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
		tab.Button.TextColor3 = Color3.fromRGB(0, 0, 0)
		tab.Glow.Transparency = 0.08

		Tween(tab.Scale, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			Scale = 1.08
		})
	end

	local function GetVisibleWindow()
		local totalTabs = #dock.Tabs
		local visibleCount = math.min(dock.VisibleTabs, totalTabs)
		local startIndex = math.floor(dock.ScrollOffset) + 1

		if totalTabs > visibleCount then
			startIndex = math.clamp(startIndex, 1, totalTabs - visibleCount + 1)
		else
			startIndex = 1
		end

		local window = {}

		for slot = 1, visibleCount do
			local tabIndex = startIndex + slot - 1
			local tab = dock.Tabs[tabIndex]

			if tab then
				table.insert(window, {
					Tab = tab,
					Index = tabIndex,
					Slot = slot
				})
			end
		end

		return window
	end

	local function RenderTabs(animated)
		if not dock.TabsVisible then
			for _, tab in ipairs(dock.Tabs) do
				if tab.Button then
					Tween(tab.Button, TweenInfo.new(0.16, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
						Position = UDim2.new(0, OrbCenterX + 14, 0.5, 0),
						BackgroundTransparency = 1,
						TextTransparency = 1,
						Rotation = 0
					})

					Tween(tab.Scale, TweenInfo.new(0.16, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
						Scale = 0.35
					})

					if tab.Outline then
						tab.Outline.Transparency = 1
					end

					tab.Glow.Transparency = 1

					if tab.NameLabel then
						Tween(tab.NameLabel, TweenInfo.new(0.14), {
							TextTransparency = 1,
							BackgroundTransparency = 1
						})
					end

					if tab.NameStroke then
						Tween(tab.NameStroke, TweenInfo.new(0.14), {
							Transparency = 1
						})
					end
				end
			end

			return
		end

		local visibleWindow = GetVisibleWindow()
		local visibleMap = {}

		for _, data in ipairs(visibleWindow) do
			visibleMap[data.Tab] = data
		end

		for _, tab in ipairs(dock.Tabs) do
			if tab.Button then
				local data = visibleMap[tab]

				if data then
					local angle = SlotAngles[data.Slot] or 0
					local isActive = dock.CurrentTab == tab

					tab.Button.Visible = true

					Tween(tab.Button, TweenInfo.new(animated and 0.24 or 0.16, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						Position = GetTabPositionByAngle(angle),
						Rotation = angle * 0.08,
						BackgroundTransparency = 0,
						TextTransparency = 0
					})

					if tab.NameLabel then
						tab.NameLabel.Text = tab.Name
						tab.NameLabel.Visible = true

						Tween(tab.NameLabel, TweenInfo.new(animated and 0.22 or 0.14), {
							TextTransparency = 0,
							BackgroundTransparency = 0.12
						})
					end

					if tab.NameStroke then
						Tween(tab.NameStroke, TweenInfo.new(animated and 0.22 or 0.14), {
							Transparency = 0.72
						})
					end

					if tab.Outline then
						tab.Outline.Transparency = 0.55
					end

					if isActive then
						ForceActiveTab(tab)
					else
						tab.Button.BackgroundColor3 = Library.Theme.TabDark
						tab.Button.TextColor3 = Library.Theme.Text
						tab.Glow.Transparency = 1

						Tween(tab.Scale, TweenInfo.new(animated and 0.24 or 0.16, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
							Scale = 1
						})
					end
				else
					local tabIndex = table.find(dock.Tabs, tab) or 1
					local startIndex = math.floor(dock.ScrollOffset) + 1

					local hideAngle
					if tabIndex < startIndex then
						hideAngle = SlotAngles[1] - 36
					else
						hideAngle = SlotAngles[#SlotAngles] + 36
					end

					Tween(tab.Button, TweenInfo.new(0.18), {
						Position = GetTabPositionByAngle(hideAngle),
						BackgroundTransparency = 1,
						TextTransparency = 1,
						Rotation = hideAngle * 0.08
					})

					if tab.NameLabel then
						Tween(tab.NameLabel, TweenInfo.new(0.14), {
							TextTransparency = 1,
							BackgroundTransparency = 1
						})
					end

					if tab.NameStroke then
						Tween(tab.NameStroke, TweenInfo.new(0.14), {
							Transparency = 1
						})
					end

					Tween(tab.Scale, TweenInfo.new(0.16), {
						Scale = 0.55
					})

					if tab.Outline then
						tab.Outline.Transparency = 1
					end

					tab.Glow.Transparency = 1
				end
			end
		end
	end

	local function ScrollTabs(direction)
		if #dock.Tabs <= dock.VisibleTabs then
			return
		end

		local maxOffset = GetMaxScrollOffset()

		local newOffset = math.clamp(
			dock.ScrollOffset + direction,
			0,
			maxOffset
		)

		if newOffset == dock.ScrollOffset then
			return
		end

		dock.ScrollOffset = newOffset
		RenderTabs(true)
	end

	local function EnsureTabCentered(tab)
		local tabIndex = table.find(dock.Tabs, tab)
		if not tabIndex then
			return
		end

		local maxOffset = GetMaxScrollOffset()
		local centerSlot = math.floor(dock.VisibleTabs / 2)
		local wantedOffset = (tabIndex - 1) - centerSlot

		dock.ScrollOffset = math.clamp(wantedOffset, 0, maxOffset)
	end

	local function SelectTab(tab)
		if dock.CurrentTab == tab and dock.PanelOpen then
			dock.LastOpenTab = nil
			ClosePanel()
			RenderTabs(false)
			return
		end

		EnsureTabCentered(tab)
		RenderTabs(true)

		OpenPanel()

		dock.CurrentTab = tab
		dock.LastOpenTab = tab
		PageTitle.Text = tab.Name
		ClearContent()

		for _, other in ipairs(dock.Tabs) do
			if other.Button then
				other.Button.BackgroundColor3 = Library.Theme.TabDark
				other.Button.TextColor3 = Library.Theme.Text
				other.Glow.Transparency = 1

				Tween(other.Scale, TweenInfo.new(0.14), {
					Scale = 1
				})
			end
		end

		ForceActiveTab(tab)

		for _, createItem in ipairs(tab.Items) do
			createItem()
		end

		RenderTabs(false)
	end

	local function SetTabsVisible(value)
		dock.TabsVisible = value
		RenderTabs(true)
	end

	local function ToggleTabs()
		local newState = not dock.TabsVisible

		if not newState then
			if dock.CurrentTab and dock.PanelOpen then
				dock.LastOpenTab = dock.CurrentTab
			end

			SetTabsVisible(false)
			ClosePanel()
		else
			SetTabsVisible(true)

			if dock.LastOpenTab then
				task.delay(0.12, function()
					if dock.TabsVisible and dock.LastOpenTab then
						SelectTab(dock.LastOpenTab)
					end
				end)
			end
		end
	end

	local LastWheelTime = 0
	local WheelCooldown = 0.08

	UserInputService.InputChanged:Connect(function(input)
		if not dock.TabsVisible then
			return
		end

		if input.UserInputType ~= Enum.UserInputType.MouseWheel then
			return
		end

		local now = os.clock()
		if now - LastWheelTime < WheelCooldown then
			return
		end

		local mouse = UserInputService:GetMouseLocation()
		local pos = OrbArea.AbsolutePosition
		local size = OrbArea.AbsoluteSize

		local insideOrbArea =
			mouse.X >= pos.X and
			mouse.X <= pos.X + size.X and
			mouse.Y >= pos.Y and
			mouse.Y <= pos.Y + size.Y

		if not insideOrbArea then
			return
		end

		LastWheelTime = now

		if input.Position.Z > 0 then
			ScrollTabs(-1)
		elseif input.Position.Z < 0 then
			ScrollTabs(1)
		end
	end)

				switch.Position = UDim2.new(1, -13, 0.5, 0)
				switch.Size = UDim2.new(0, 48, 0, 24)
				switch.BackgroundColor3 = state and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(35, 35, 35)
				switch.BorderSizePixel = 0
				switch.Text = ""
				switch.AutoButtonColor = false
				switch.ZIndex = 13
				switch.Parent = card

				Corner(switch, 999)
				Stroke(switch, Color3.fromRGB(255, 255, 255), 1, 0.72)

				local knob = Instance.new("Frame")
				knob.Size = UDim2.new(0, 18, 0, 18)
				knob.Position = state and UDim2.new(1, -21, 0.5, -9) or UDim2.new(0, 3, 0.5, -9)
				knob.BackgroundColor3 = state and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)
				knob.BorderSizePixel = 0
				knob.ZIndex = 14
				knob.Parent = switch

				Corner(knob, 999)

				local lastToggle = 0

				local function Set(value)
					state = value

					Tween(switch, TweenInfo.new(0.2), {
						BackgroundColor3 = state and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(35, 35, 35)
					})

					Tween(knob, TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						Position = state and UDim2.new(1, -21, 0.5, -9) or UDim2.new(0, 3, 0.5, -9),
						BackgroundColor3 = state and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)
					})

					if itemOptions.Callback then
						itemOptions.Callback(state)
					end
				end

				local function Toggle()
					local now = os.clock()
					if now - lastToggle < 0.06 then
						return
					end

					lastToggle = now
					Set(not state)
				end

				switch.MouseButton1Click:Connect(Toggle)

				card.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
						Toggle()
					end
				end)
			end)
		end

		function tab:CreateAnimationCatalog()
			local hasLoaded = false
			local isLoading = false
			local allAnimations = {}
			local filteredAnimations = {}

			table.insert(self.Items, function()
				local search = Instance.new("TextBox")
				search.Size = UDim2.new(1, 0, 0, 34)
				search.BackgroundColor3 = Color3.fromRGB(24, 24, 24)
				search.BorderSizePixel = 0
				search.ClearTextOnFocus = false
				search.PlaceholderText = "Search animation packs..."
				search.Text = ""
				search.Font = Library.Theme.Font
				search.TextColor3 = Library.Theme.Text
				search.PlaceholderColor3 = Library.Theme.Muted
				search.TextSize = 11
				search.TextXAlignment = Enum.TextXAlignment.Left
				search.ZIndex = 13
				search.Parent = Content

				Corner(search, 8)
				Padding(search, 9)
				Stroke(search, Color3.fromRGB(255, 255, 255), 1, 0.82)

				local statusCard = CreateCard(Content, 62)
				statusCard.ZIndex = 12

				local statusTitle = Instance.new("TextLabel")
				statusTitle.BackgroundTransparency = 1
				statusTitle.Position = UDim2.new(0, 13, 0, 10)
				statusTitle.Size = UDim2.new(1, -26, 0, 20)
				statusTitle.Font = Library.Theme.BoldFont
				statusTitle.Text = hasLoaded and "Animations Loaded" or "Loading Animations..."
				statusTitle.TextColor3 = Library.Theme.Text
				statusTitle.TextSize = 13
				statusTitle.TextXAlignment = Enum.TextXAlignment.Left
				statusTitle.ZIndex = 13
				statusTitle.Parent = statusCard

				local statusDesc = Instance.new("TextLabel")
				statusDesc.BackgroundTransparency = 1
				statusDesc.Position = UDim2.new(0, 13, 0, 33)
				statusDesc.Size = UDim2.new(1, -26, 0, 17)
				statusDesc.Font = Library.Theme.Font
				statusDesc.Text = hasLoaded and (tostring(#allAnimations) .. " animation packs ready.") or "Fetching catalog..."
				statusDesc.TextColor3 = Library.Theme.Muted
				statusDesc.TextSize = 10
				statusDesc.TextXAlignment = Enum.TextXAlignment.Left
				statusDesc.ZIndex = 13
				statusDesc.Parent = statusCard

				local holder = Instance.new("Frame")
				holder.Size = UDim2.new(1, 0, 0, 1)
				holder.BackgroundTransparency = 1
				holder.ZIndex = 12
				holder.Parent = Content

				local holderLayout = Instance.new("UIListLayout")
				holderLayout.Padding = UDim.new(0, 8)
				holderLayout.SortOrder = Enum.SortOrder.LayoutOrder
				holderLayout.Parent = holder

				local renderToken = 0

				local function updateHolderSize()
					task.defer(function()
						if holder and holder.Parent then
							holder.Size = UDim2.new(1, 0, 0, holderLayout.AbsoluteContentSize.Y)
						end
					end)
				end

				local function clearHolder()
					renderToken += 1

					for _, child in ipairs(holder:GetChildren()) do
						if not child:IsA("UIListLayout") then
							child:Destroy()
						end
					end

					updateHolderSize()
				end

				local function filterAnimations()
					local term = string.lower(search.Text or "")
					filteredAnimations = {}

					for _, item in ipairs(allAnimations) do
						local name = string.lower(tostring(item.Name or ""))
						local id = tostring(item.Id or "")

						if term == "" or name:find(term, 1, true) or id:find(term, 1, true) then
							table.insert(filteredAnimations, item)
						end
					end
				end

				local function createAnimationCard(item, index, token)
					local card = Instance.new("TextButton")
					card.Name = tostring(item.Name or "Animation")
					card.Size = UDim2.new(1, 0, 0, 104)
					card.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
					card.BorderSizePixel = 0
					card.AutoButtonColor = false
					card.Text = ""
					card.ZIndex = 12
					card.Parent = holder

					Corner(card, 12)
					local cardGradient = Gradient(card, Library.Theme.CardDark, Library.Theme.CardLight, 68)
					Stroke(card, Color3.fromRGB(255, 255, 255), 1, 0.72)

					local viewport, world, stateLabel = CreateViewport(card)

					local title = Instance.new("TextLabel")
					title.BackgroundTransparency = 1
					title.Position = UDim2.new(0, 104, 0, 13)
					title.Size = UDim2.new(1, -118, 0, 22)
					title.Font = Library.Theme.BoldFont
					title.Text = tostring(item.Name or "Animation")
					title.TextColor3 = Library.Theme.Text
					title.TextSize = 14
					title.TextXAlignment = Enum.TextXAlignment.Left
					title.ZIndex = 13
					title.Parent = card

					local desc = Instance.new("TextLabel")
					desc.BackgroundTransparency = 1
					desc.Position = UDim2.new(0, 104, 0, 38)
					desc.Size = UDim2.new(1, -118, 0, 34)
					desc.Font = Library.Theme.Font
					desc.Text = "ID: " .. tostring(item.Id or "?")
					desc.TextColor3 = Library.Theme.Muted
					desc.TextSize = 10
					desc.TextWrapped = true
					desc.TextXAlignment = Enum.TextXAlignment.Left
					desc.TextYAlignment = Enum.TextYAlignment.Top
					desc.ZIndex = 13
					desc.Parent = card

					local action = Instance.new("TextLabel")
					action.AnchorPoint = Vector2.new(1, 1)
					action.Position = UDim2.new(1, -13, 1, -12)
					action.Size = UDim2.new(0, 64, 0, 22)
					action.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
					action.BorderSizePixel = 0
					action.Font = Library.Theme.BoldFont
					action.Text = "APPLY"
					action.TextColor3 = Color3.fromRGB(0, 0, 0)
					action.TextSize = 9
					action.ZIndex = 13
					action.Parent = card

					Corner(action, 4)

					card.MouseEnter:Connect(function()
						cardGradient.Color = ColorSequence.new({
							ColorSequenceKeypoint.new(0, Library.Theme.CardHover),
							ColorSequenceKeypoint.new(1, Color3.fromRGB(80, 80, 80))
						})
					end)

					card.MouseLeave:Connect(function()
						cardGradient.Color = ColorSequence.new({
							ColorSequenceKeypoint.new(0, Library.Theme.CardDark),
							ColorSequenceKeypoint.new(1, Library.Theme.CardLight)
						})
					end)

					card.MouseButton1Click:Connect(function()
						ApplyAnimationPack(item)
					end)

					task.delay(index * 0.12, function()
						if renderToken ~= token or not card.Parent then
							return
						end

						stateLabel.Text = "Loading"

						local dummy = CreateGrayDummy(world)

						if not dummy then
							stateLabel.Text = "No Dummy"
							return
						end

						local mappings = ResolveAnimationMappings(item)
						local order = {"idle", "walk", "run", "jump", "fall", "climb", "swim"}

						task.spawn(function()
							while renderToken == token and card.Parent and dummy.Parent do
								for _, category in ipairs(order) do
									if renderToken ~= token or not card.Parent or not dummy.Parent then
										break
									end

									local chosen = nil

									for _, mapping in ipairs(mappings) do
										if mapping.category == category then
											chosen = mapping
											break
										end
									end

									if chosen then
										PlayPreviewAnimation(
											dummy,
											chosen.animationId,
											stateLabel,
											category:sub(1, 1):upper() .. category:sub(2),
											2
										)
									else
										stateLabel.Text = category:sub(1, 1):upper() .. category:sub(2)
										task.wait(2)
									end
								end
							end
						end)
					end)
				end

				local function renderAnimations()
					clearHolder()
					filterAnimations()

					local token = renderToken

					for index, item in ipairs(filteredAnimations) do
						createAnimationCard(item, index, token)
					end

					updateHolderSize()
				end

				search:GetPropertyChangedSignal("Text"):Connect(function()
					if hasLoaded then
						renderAnimations()
					end
				end)

				if hasLoaded then
					renderAnimations()
					return
				end

				if isLoading then
					statusTitle.Text = "Loading Animations..."
					statusDesc.Text = "Already fetching catalog."
					return
				end

				isLoading = true
				Library:Notify("Loading animations...", Color3.fromRGB(255, 255, 255), "info")

				task.spawn(function()
					allAnimations = FetchAllAnimationPacks()
					hasLoaded = true
					isLoading = false

					if statusTitle and statusTitle.Parent then
						statusTitle.Text = "Animations Loaded"
					end

					if statusDesc and statusDesc.Parent then
						statusDesc.Text = tostring(#allAnimations) .. " animation packs loaded."
					end

					renderAnimations()

					Library:Notify("Loaded " .. tostring(#allAnimations) .. " animations.", Color3.fromRGB(255, 255, 255), "check")
				end)
			end)
		end

		table.insert(self.Tabs, tab)
		RenderTabs(false)

		return tab
	end

	function dock:Destroy()
		Root:Destroy()
	end

	function dock:Notify(...)
		Library:Notify(...)
	end

	return dock
end
